    "Reads a `.docx` and builds a structured index of paragraphs and tables."

    "# ─────────────────────────────────────────────────────────────────────────────\n",
    "#  HELPERS\n",
    "# ─────────────────────────────────────────────────────────────────────────────\n",
    "\n",
    "def normalise(text: str) -> str:\n",
    "    \"\"\"Lowercase + collapse whitespace.\"\"\"\n",
    "    return re.sub(r\"\\s+\", \" \", text).strip().lower()\n",
    "\n",
    "\n",
    "HEADING_STYLE_RE = re.compile(r\"(?i)heading\\s*(\\d+)\")\n",
    "\n",
    "# Matches numbered headings even without a Word Heading style:\n",
    "#   \"1 Introduction\", \"2.3 Methodology\", \"A.1 Appendix\"\n",
    "INLINE_HEADING_RE = re.compile(\n",
    "    r\"^(?P<num>(?:\\d+\\.){0,3}\\d+|[A-Z]\\.\\d+)[\\s\\.\\-\\u2013\\u2014:]+(?P<title>[A-Z][^\\n]{2,80})$\"\n",
    ")\n",
    "\n",
    "TABLE_CAPTION_RE = re.compile(\n",
    "    r\"(?i)(?:table|exhibit)\\s+(?P<ref>[\\w\\-\\.]+)[:\\s\\-\\u2013\\u2014]*(?P<title>[^\\n]{0,80})\"\n",
    ")\n",
    "\n",
    "\n",
    "def is_heading(para) -> tuple:\n",
    "    \"\"\"\n",
    "    Returns (is_heading: bool, level: int, number: str, title: str)\n",
    "    Detects both Word Heading styles and inline numbered headings.\n",
    "    \"\"\"\n",
    "    style_name = para.style.name if para.style else \"\"\n",
    "    m_style = HEADING_STYLE_RE.match(style_name)\n",
    "    if m_style:\n",
    "        level = int(m_style.group(1))\n",
    "        # Try to parse a number out of the text itself\n",
    "        text = para.text.strip()\n",
    "        m_inline = INLINE_HEADING_RE.match(text)\n",
    "        if m_inline:\n",
    "            return True, level, m_inline.group(\"num\"), m_inline.group(\"title\")\n",
    "        return True, level, \"\", text\n",
    "\n",
    "    # Fallback: Bold paragraph that looks like a numbered heading\n",
    "    text = para.text.strip()\n",
    "    if text:\n",
    "        m_inline = INLINE_HEADING_RE.match(text)\n",
    "        if m_inline:\n",
    "            # Check if the paragraph is bold\n",
    "            runs_bold = [r.bold for r in para.runs if r.text.strip()]\n",
    "            if runs_bold and all(runs_bold):\n",
    "                return True, 99, m_inline.group(\"num\"), m_inline.group(\"title\")\n",
    "\n",
    "    return False, 0, \"\", \"\"\n",
    "\n",
    "\n",
    "def estimate_page(element_index: int, total_elements: int, total_pages_hint: int = 1) -> str:\n",
    "    \"\"\"\n",
    "    Word documents do NOT store page numbers in paragraph XML.\n",
    "    We estimate page based on paragraph index position.\n",
    "    For accurate page numbers, convert to PDF first.\n",
    "    \"\"\"\n",
    "    if total_elements == 0:\n",
    "        return \"~1\"\n",
    "    ratio = element_index / total_elements\n",
    "    est = max(1, round(ratio * total_pages_hint))\n",
    "    return f\"~{est}\"\n",
    "\n",

    "# ─────────────────────────────────────────────────────────────────────────────\n",
    "#  DOCUMENT INDEXER\n",
    "#  Builds a flat list of \"blocks\" from a Word document:\n",
    "#    - paragraph blocks\n",
    "#    - table blocks (one per table)\n",
    "#  Each block carries section context and element index.\n",
    "# ─────────────────────────────────────────────────────────────────────────────\n",
    "\n",
    "def index_document(docx_path: str) -> dict:\n",
    "    \"\"\"\n",
    "    Returns:\n",
    "      {\n",
    "        'doc'      : Document object,\n",
    "        'blocks'   : [ block_dict, ... ],\n",
    "        'sections' : [ {section_num, section_name, block_idx}, ... ],\n",
    "        'total'    : int   # total block count\n",
    "      }\n",
    "    \"\"\"\n",
    "    doc = Document(docx_path)\n",
    "\n",
    "    # ── Estimate total pages from document core properties ───────────────────\n",
    "    try:\n",
    "        total_pages = doc.core_properties.revision or 20   # rough fallback\n",
    "    except Exception:\n",
    "        total_pages = 20\n",
    "\n",
    "    # ── Walk document body in order (paragraphs AND tables interleaved) ───────\n",
    "    # python-docx doc.paragraphs skips table cells; we need doc.element body\n",
    "    body = doc.element.body\n",
    "    blocks = []\n",
    "    section_stack = [{\"section_num\": \"—\", \"section_name\": \"(preamble)\"}]\n",
    "    table_counter = 0\n",
    "\n",
    "    # We'll build a map from element → block later for section lookup\n",
    "    para_map  = {p._element: p for p in doc.paragraphs}\n",
    "    table_map = {t._element: t for t in doc.tables}\n",
    "\n",
    "    for child in body:\n",
    "        tag = child.tag.split(\"}\")[-1] if \"}\" in child.tag else child.tag\n",
    "\n",
    "        if tag == \"p\":                          # paragraph\n",
    "            para = para_map.get(child)\n",
    "            if para is None:\n",
    "                continue\n",
    "            text = para.text.strip()\n",
    "\n",
    "            # Check for table caption BEFORE the table\n",
    "            cap_m = TABLE_CAPTION_RE.search(text)\n",
    "            if cap_m:\n",
    "                pending_table_caption = f\"Table {cap_m.group('ref')}\"\n",
    "                if cap_m.group('title').strip():\n",
    "                    pending_table_caption += f\" – {cap_m.group('title').strip()}\"\n",
    "            else:\n",
    "                pending_table_caption = None\n",
    "\n",
    "            # Heading detection\n",
    "            ok, level, num, title = is_heading(para)\n",
    "            if ok and title:\n",
    "                section_stack = [s for s in section_stack if s.get(\"level\", 99) < level]\n",
    "                section_stack.append({\n",
    "                    \"section_num\" : num or title[:6],\n",
    "                    \"section_name\": title,\n",
    "                    \"level\"       : level,\n",
    "                })\n",
    "\n",
    "            blocks.append({\n",
    "                \"kind\"        : \"paragraph\",\n",
    "                \"text\"        : text,\n",
    "                \"style\"       : para.style.name if para.style else \"\",\n",
    "                \"section_num\" : section_stack[-1][\"section_num\"],\n",
    "                \"section_name\": section_stack[-1][\"section_name\"],\n",
    "                \"table_ref\"   : pending_table_caption or \"—\",\n",
    "                \"block_idx\"   : len(blocks),\n",
    "            })\n",
    "\n",
    "        elif tag == \"tbl\":                      # table\n",
    "            tbl = table_map.get(child)\n",
    "            if tbl is None:\n",
    "                continue\n",
    "            table_counter += 1\n",
    "\n",
    "            # Collect rows as list of list of strings\n",
    "            rows_data = []\n",
    "            for row in tbl.rows:\n",
    "                row_cells = [c.text.strip() for c in row.cells]\n",
    "                rows_data.append(row_cells)\n",
    "\n",
    "            # Infer table caption from last paragraph block\n",
    "            last_cap = \"—\"\n",
    "            for b in reversed(blocks):\n",
    "                if b[\"kind\"] == \"paragraph\" and b[\"table_ref\"] != \"—\":\n",
    "                    last_cap = b[\"table_ref\"]\n",
    "                    break\n",
    "                if b[\"kind\"] == \"paragraph\" and b[\"text\"]:\n",
    "                    # Check if immediately preceding para is a table caption\n",
    "                    cap_m2 = TABLE_CAPTION_RE.search(b[\"text\"])\n",
    "                    if cap_m2:\n",
    "                        last_cap = f\"Table {cap_m2.group('ref')}\"\n",
    "                        if cap_m2.group('title').strip():\n",
    "                            last_cap += f\" – {cap_m2.group('title').strip()}\"\n",
    "                    break\n",
    "\n",
    "            default_ref = f\"Table {table_counter}\"\n",
    "            table_ref = last_cap if last_cap != \"—\" else default_ref\n",
    "\n",
    "            blocks.append({\n",
    "                \"kind\"        : \"table\",\n",
    "                \"rows\"        : rows_data,\n",
    "                \"table_ref\"   : table_ref,\n",
    "                \"section_num\" : section_stack[-1][\"section_num\"],\n",
    "                \"section_name\": section_stack[-1][\"section_name\"],\n",
    "                \"block_idx\"   : len(blocks),\n",
    "            })\n",
    "\n",
    "    total = len(blocks)\n",
    "    # Back-fill estimated page numbers\n",

    "# ─────────────────────────────────────────────────────────────────────────────\n",
    "#  EXTRACTION FUNCTIONS\n",
    "# ─────────────────────────────────────────────────────────────────────────────\n",
    "\n",
    "def search_tables(blocks: list, aliases: list) -> list:\n",
    "    \"\"\"\n",
    "    Search table blocks for a row whose first cell matches any alias.\n",
    "    Returns list of hit dicts.\n",
    "    \"\"\"\n",
    "    norm_aliases = [normalise(a) for a in aliases]\n",
    "    hits = []\n",
    "\n",
    "    for block in blocks:\n",
    "        if block[\"kind\"] != \"table\":\n",
    "            continue\n",
    "        for row in block[\"rows\"]:\n",
    "            if not row:\n",
    "                continue\n",
    "            key = normalise(row[0])\n",
    "            matched = any(\n",
    "                alias in key or key in alias\n",
    "                for alias in norm_aliases\n",
    "            )\n",
    "            if matched:\n",
    "                value_cells = [c.strip() for c in row[1:] if c.strip()]\n",
    "                value = \" | \".join(value_cells) if value_cells else \"(empty)\"\n",
    "                hits.append({\n",
    "                    \"value\"        : value,\n",
    "                    \"location_type\": \"Table\",\n",
    "                    \"table_ref\"    : block[\"table_ref\"],\n",
    "                    \"section_num\"  : block[\"section_num\"],\n",
    "                    \"section_name\" : block[\"section_name\"],\n",
    "                    \"est_page\"     : block[\"est_page\"],\n",
    "                })\n",
    "    return hits\n",
    "\n",
    "\n",
    "def search_paragraphs(blocks: list, aliases: list, context_chars: int = 350) -> list:\n",
    "    \"\"\"\n",
    "    Search paragraph blocks for label-value patterns:\n",
    "      \"Label: value text...\"\n",
    "      \"Label — value text...\"\n",
    "    Returns list of hit dicts.\n",
    "    \"\"\"\n",
    "    hits = []\n",
    "    seen = set()\n",
    "\n",
    "    # Build compiled patterns once\n",
    "    patterns = []\n",
    "    for alias in aliases:\n",
    "        pat = re.compile(\n",
    "            rf\"(?i)(?:^|[\\s])(?P<label>{re.escape(alias)})\\s*[:\\u2013\\u2014\\-]\\s*(?P<value>[^\\n]{{2,{context_chars}}})\",\n",
    "        )\n",
    "        patterns.append(pat)\n",
    "\n",
    "    for block in blocks:\n",
    "        if block[\"kind\"] != \"paragraph\" or not block[\"text\"]:\n",
    "            continue\n",
    "        text = block[\"text\"]\n",
    "\n",
    "        for pat in patterns:\n",
    "            m = pat.search(text)\n",
    "            if m:\n",
    "                value = m.group(\"value\").strip()[:350]\n",
    "                dedup_key = (normalise(value), block[\"section_num\"])\n",
    "                if dedup_key in seen:\n",
    "                    continue\n",
    "                seen.add(dedup_key)\n",
    "                hits.append({\n",
    "                    \"value\"        : value,\n",
    "                    \"location_type\": \"Section\",\n",
    "                    \"table_ref\"    : \"—\",\n",
    "                    \"section_num\"  : block[\"section_num\"],\n",
    "                    \"section_name\" : block[\"section_name\"],\n",
    "                    \"est_page\"     : block[\"est_page\"],\n",
    "                })\n",
    "    return hits\n",
    "\n",
    "\n",
    "def extract_field(field: dict, blocks: list) -> list:\n",
    "    \"\"\"\n",
    "    Main dispatch: tries table search and/or paragraph search\n",
    "    depending on field[\"location\"].\n",
    "    \"\"\"\n",
    "    aliases = field[\"aliases\"]\n",
    "    loc     = field[\"location\"]\n",
    "    hits    = []\n",
    "\n",
    "    if loc in (\"table\", \"any\"):\n",
    "        hits = search_tables(blocks, aliases)\n",
    "\n",
    "    if not hits and loc in (\"section\", \"any\"):\n",
    "        hits = search_paragraphs(blocks, aliases)\n",
    "\n",
    "    # If table-preferred but nothing found, try paragraphs as fallback\n",
    "    if not hits and loc == \"table\":\n",
    "        hits = search_paragraphs(blocks, aliases)\n",
    "\n",
    "    return hits\n",
    "\n",
    "\n",
    "print(\"Extraction functions defined ✓\")"
   ]
